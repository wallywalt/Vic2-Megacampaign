	id = 160011
	title = "EVTNAME160011" #Taiping Missionaries in $STATENAME$
	desc = "EVTDESC160011"
	picture = "taiping"
	
	trigger = {
		year = 1856
		tag = MNG
		check_variable = {
			which = taiping_provinces
			value = 5
		}
		NOT = { has_country_flag = heavenly_kingdom }

		#NOT = {
			#has_global_flag = taiping_has_happened
		#}
	}
	
	mean_time_to_happen = {
		months = 180
	}
	
	option = {		
		name = "EVTOPT160011"
		set_variable = {
			which = taiping_provinces
			value = 0
		}
		set_country_flag = heavenly_kingdom
		
		random_province = {
			limit = {
				controlled_by = MNG
				any_neighbor_province = { owned_by = FRA }
			}
			any_pop = {
				limit = { is_primary_culture = no }
				militancy = 10
			}
			add_core = TPG
		}
	}
}
